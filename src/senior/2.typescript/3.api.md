# 三.composition-api

## 1.案例

```vue
<template>
  <button @click="increment">
    当前是: {{ state.count }}, 双倍是: {{ state.double }}
  </button>
</template>

<script>
import { computed, reactive `} from "vue"
export default {
  setup() {
    const state = reactive({
      count: 1,
      double: computed(() => state.count *2),
    })

    function increment() {
      state.count++
    }
    return {
      state,
      increment,
    }
  },
}
</script>
```
