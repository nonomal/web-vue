# 构建部署

## 5.

::: demo

```vue
<template>
  <div ref="map" style="height: 400px"></div>
</template>

<script>
export default {
  data() {
    return {};
  },
  mounted() {
    let target = [{ name: "南海诸岛", value: 0 }];
    let result = [];
    china.features.forEach((item) => {
      let data;
      result.forEach((i) => {
        if (i.province === item.properties.name) {
          data = { name: i.province, value: i.machineNumbers };
        }
      });
      target.push(data ? data : { name: item.properties.name, value: 0 });
    });
    let option = {
      title: {
        text: "所有车辆分布",
        left: 20,
        top: 26,
        textStyle: {
          fontSize: 22,
        },
      },
      tooltip: {
        trigger: "item",
        formatter: "{c}辆",
      },
      visualMap: {
        min: 0,
        max: 100,
        text: ["高", "低"],
        realtime: false,
        calculable: false,
        orient: "horizontal",
        inRange: {
          color: ["#e9f3fc", "#90bcf1", "#3885e2"],
        },
        itemWidth: 16,
        itemHeight: 50,
        left: 40,
        bottom: 62,
      },
      series: [
        {
          name: "中国地图",
          type: "map",
          map: "china",
          label: {
            show: true,
            formatter: "{c}",
          },
          data: target,
          left: 34,
          top: 128,
          zoom: 1.1,
        },
      ],
    };
    echarts.registerMap("china", china);
    let charts = echarts.init(this.$refs.map);
    charts.setOption(option);
  },
};
</script>
```
